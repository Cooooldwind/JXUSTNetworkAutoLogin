### 2.2 扩展自启动功能

**修改 `core/autostart.py`:**

```python
import os
import sys
import winreg
import subprocess

APP_NAME = "JXUSTNetLogin"

# 现有注册表自启动代码保持不变

# 任务计划程序自启动
def enable_task_scheduler() -> bool:
    """启用任务计划程序自启动"""
    try:
        cmd = f'schtasks.exe /Create /TN "{APP_NAME}" /TR "{_launch_command()}" /SC ONLOGON /RL HIGHEST /F'
        subprocess.run(cmd, shell=True, check=True, capture_output=True, text=True)
        return True
    except Exception:
        return False

def disable_task_scheduler() -> bool:
    """禁用任务计划程序自启动"""
    try:
        cmd = f'schtasks.exe /Delete /TN "{APP_NAME}" /F'
        subprocess.run(cmd, shell=True, check=True, capture_output=True, text=True)
        return True
    except Exception:
        return True  # 如果任务不存在，也视为成功

def is_task_scheduler_enabled() -> bool:
    """检查任务计划程序自启动是否启用"""
    try:
        cmd = f'schtasks.exe /Query /TN "{APP_NAME}" /FO CSV'
        result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
        return result.returncode == 0
    except Exception:
        return False

# 系统服务自启动
def enable_service() -> bool:
    """启用系统服务自启动"""
    try:
        exe_path = sys.executable if getattr(sys, "frozen", False) else f'{sys.executable} {os.path.abspath(sys.argv[0])}'
        cmd = f'sc.exe create "{APP_NAME}" binPath= "{exe_path} --silent" start= auto displayname= "{APP_NAME}"'
        subprocess.run(cmd, shell=True, check=True, capture_output=True, text=True)
        return True
    except Exception:
        return False

def disable_service() -> bool:
    """禁用系统服务自启动"""
    try:
        # 先停止服务（如果正在运行）
        subprocess.run(f'sc.exe stop "{APP_NAME}"', shell=True, capture_output=True, text=True)
        # 删除服务
        subprocess.run(f'sc.exe delete "{APP_NAME}"', shell=True, check=True, capture_output=True, text=True)
        return True
    except Exception:
        return True  # 如果服务不存在，也视为成功

def is_service_enabled() -> bool:
    """检查系统服务自启动是否启用"""
    try:
        cmd = f'sc.exe query "{APP_NAME}"'
        result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
        return result.returncode == 0
    except Exception:
        return False
```

```python
def _init_autostart_states(self):
    from core import autostart
    
    # 初始化注册表自启动状态

    # 初始化任务计划程序自启动状态
    self.auto_start_task.setChecked(autostart.is_task_scheduler_enabled())
    
    # 初始化系统服务自启动状态
    self.auto_start_service.setChecked(autostart.is_service_enabled())

# 更新 _toggle_autostart 方法
def _toggle_autostart(self, autostart_type):
    from core import autostart
    enabled = getattr(self, f"auto_start_{autostart_type}").isChecked()
    
    success = False
    if autostart_type == "registry":
        success = autostart.enable_autostart() if enabled else autostart.disable_autostart()
    elif autostart_type == "task_scheduler":
        success = autostart.enable_task_scheduler() if enabled else autostart.disable_task_scheduler()
    elif autostart_type == "service":
        success = autostart.enable_service() if enabled else autostart.disable_service()
    
    # 更新配置
    self.config["auto_start"][autostart_type] = enabled and success
    
    # 显示提示
    type_names = {
        "registry": "注册表自启动",
        "task_scheduler": "任务计划程序自启动",
        "service": "系统服务自启动"
    }
    self.show_info(f"{type_names[autostart_type]}{'已开启' if success else '开启失败'}", success)
    
    from core import config as cfg
    cfg.save_config(self.config)
```
